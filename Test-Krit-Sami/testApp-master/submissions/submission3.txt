const fs = require('fs');

process.stdin.resume();
process.stdin.setEncoding('utf-8');

let inputString = '';
let currentLine = 0;

    inputString += inputStdin;
});

    inputString = inputString.trim().split('\n').map(str => str.trim());

    main();
});
